\section*{Internships}
\begin{itemize}

\item \begin{project}{Research Intern, NYU Tandon}{May '19 -- July '19}{Prof.\ Pawe≈Ç Korus, Prof.\ Nasir Memon}%chktex 8
    \item Researched \textit{robust image hashes} that are immune to typical image transformations, measured by \href{https://en.wikipedia.org/wiki/Hamming\_distance}{Hamming distances}, while being sensitive to malicious image edits such as face swaps, deep fakes, object addition/deletion.
    \item Developed a framework for testing against transformations like compression, contrast changes, blurring, warping.
    \item Constructed various novel network architectures that analyse images in both the spatial and frequency domains.
    \item Trained models on the \href{https://dl.acm.org/citation.cfm?id=1756042}{\textit{triplet loss}} along with \href{https://arxiv.org/abs/1503.03832}{mining} of examples for improved training.
    \item Tested the networks against \textit{adversarial attacks} such as \href{https://arxiv.org/abs/1412.6572}{FGSM}, \href{https://arxiv.org/abs/1706.06083}{Projected Gradient Descent}, \href{https://arxiv.org/abs/1712.04248}{Boundary Attack}.

\item \begin{project}[https://github.com/rharish101/Plasticity-Networks]{RTE Intern, NYU Tandon}{May '18 -- July '18}{Prof.\ Yao Wang}%chktex 8
    \item Researched \href{https://arxiv.org/abs/1804.02464}{\textit{differentiable plasticity}} for \textit{domain transfer} in images using Convolutional Neural Networks (CNNs).
    \item Improved efficiency in the temporal update rule for the Hebbian weights for convolutional layers.
    \item Tested various ResNet-based architectures on the Street View House Numbers (SVHN) dataset.
    \item Improved accuracy for smaller models using plasticity, when adapting models pre-trained on SVHN for MNIST.\@
\end{project}

    \item Developed a \href{https://arxiv.org/abs/1301.3781}{Word2Vec} model to identify duplicate documents using \href{http://proceedings.mlr.press/v37/kusnerb15.html}{Word Mover's Distance} on word vectors.
